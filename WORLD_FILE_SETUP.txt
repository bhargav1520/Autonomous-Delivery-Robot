# ===== STEP 1: Make the robot a Supervisor =====
# Find your Robot node and add 'supervisor TRUE':

DEF HEXAPOD_ROBOT Robot {
  translation 0 0.3 0
  rotation 0 0 1 0
  children [
    # ... existing robot definition ...
  ]
  supervisor TRUE          # <--- ADD THIS LINE
  controller "delivery_robot"
  controllerArgs "webots_home=/dev/null"
}


# ===== STEP 2: Add Display device to robot =====
# Inside the robot's 'children' array, add this Display node:

Display {
  translation 0 0 0
  rotation 0 0 1 0
  width 200
  height 200
  children [
    Shape {
      appearance Appearance {
        material Material {
          emissiveColor 0 0 0
        }
      }
      geometry Box {
        size 0.1 0.001 0.1
      }
    }
  ]
}


# ===== STEP 3: Add 4 Box nodes to the world (NOT inside robot) =====
